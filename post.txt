create table flowers (
id int generated by default as identity primary key,
name text not null,
status text not null check(status in ('В наличии', 'Закончилось')),
price int not null,
discount int not null
);


create table users (
id int generated by default as identity primary key,
last_name text not null,
first_name text not null,
email text,
basket_id int not null
);

create table basket (
id int generated by default as identity primary key,
id_flowers int not null,
status text not null check(status in ('Оформлено', 'Не оформлено'))
id_user int not null
);

alter table users
add foreign key (basket_id) references basket(id);

alter table basket
add foreign key (id_flowers) references flowers(id);

alter table basket
add foreign key (id_user) references users(id);
