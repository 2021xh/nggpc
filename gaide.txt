Создаем .py файл пустой
рядом должен лежать готовый конструктор .ui

В терминале пишем pyuic6 untitled.ui (название конструктора) -o untitled.py (название .py файла, который высрется)
pyuic6 untitled.ui -o untitled.py


в чистом .py хуярим:
import sys 
from PyQt6.QtWidgets import QApplication, QMainWindow
from intitled import UiMainWindow        intitled - это название .py, который высран из .ui UiMainWindow - это класс в том же .py 

class MainWindow(QMainWindow, Ui_MainWindow):
    def __init__(self):
        super().__init__()
        self.setupUi(self)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = MainWindow()
    window.show()
    sys.exit(app.exec())
